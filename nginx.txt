server {
        listen [::]:80;
        listen 80;
        server_name docs.galaxroom.org;
        root /var/www/somedir;
        index /index.html;
        error_page 404 /404.html;
        server_tokens off;
        add_header X-Frame-Options "SAMEORIGIN";
        location / {
                try_files $uri $uri/ =404;
        }
        location = / {
                return 302 /docs;
        }
}
